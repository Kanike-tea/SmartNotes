import matplotlib.pyplot as plt

# Sample data
categories = ['Category A', 'Category B', 'Category C', 'Category D']
values = [10, 15, 7, 12]

# Create a bar plot
plt.bar (categories, values, color='green')

# Add labels and title
plt.xlabel ('Categories')
plt.ylabel ('Values')
plt.title('Bar Plot Example')

# Display the plot

plt.show()
