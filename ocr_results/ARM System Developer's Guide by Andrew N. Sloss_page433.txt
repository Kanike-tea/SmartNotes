#include <stdio.h>
#include <time.h>

void cache RRtest(int times, int numset)

{

clock_t count;

printf("Round Robin test size = %d\r\n", numset) ;

enableRoundRobin() ;
cleanFlushCache() ;
count = clock();

readSet (times, numset) ;
count = clock() - count;

printf("Round Robin enabled = %.2f seconds\r\n",
(float) count /CLOCKS_PER_SEC);

enableRandom() ;
cleanFlushCache() ;
count = clock();
readSet (times, numset);
count = clock() - count;

printf("Random enabled = %.2f seconds\r\n\r\n",
(float) count /CLOCKS_PER_SEC);

}
